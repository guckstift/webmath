a*x² + b*x + c = 0

x² + b/a*x + c/a = 0

x² + b/a*x + c/a + (b/(2*a))² = (b/(2*a))²

x² + b/a*x + (b/(2*a))² = (b/(2*a))² - c/a

x² + 2*(b/(2*a))*x + (b/(2*a))² = (b/(2*a))² - c/a ;

(x + b/(2*a))² = (b/(2*a))² - c/a ;

x + b/(2*a) = sqrt( (b/(2*a))² - c/a )

x = sqrt( (b/(2*a))² - c/a ) - b/(2*a)
